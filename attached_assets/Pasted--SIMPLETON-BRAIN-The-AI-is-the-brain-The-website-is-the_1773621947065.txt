/**
* SIMPLETON BRAIN™
* The AI is the brain. The website is the body.
*
* This context gives the AI awareness of everything happening in the app.
* Any component can update the brain's awareness, and any component can
* invoke the brain to take action.
*/
import { createContext, useContext, useState, useCallback, useRef, ReactNode } from "react";
// Everything the Brain can know about the current state of the app
export interface BrainAwareness {
// Where the user is
currentPage: string;
// Calculator state
calculator?: {
display: string;
selectedMetal: string;
selectedPurity: number;
selectedKarat: number;
selectedUnit: string;
livePrice: number;
loanPrice: number;
sellPrice: number;
priceMode: "live" | "custom";
scrapItems?: Array<{ weight: number; karat: string; value: number }>;
lastCalculation?: number;
};
// Diamond state
diamonds?: {
selectedCarat?: number;
selectedClarity?: string;
selectedColor?: string;
selectedCut?: string;
estimatedValue?: number;
};
// Database state
database?: {
searchQuery?: string;
selectedCoin?: string;
selectedMetal?: string;
viewingCoin?: { name: string; weight: number; purity: number; meltValue: number };
};
// Market awareness
market?: {
goldPrice: number;
silverPrice: number;
platinumPrice: number;
lastUpdated?: string;
};
// User context
user?: {
isLoggedIn: boolean;
portfolioValue?: number;
};
// What the user just did (for proactive responses)
lastAction?: {
type: string;
detail: string;
timestamp: number;
};
}
// Things the Brain can DO — commands it can send back to the app
export interface BrainAction {
type:
| "SET_CALCULATOR_METAL"
| "SET_CALCULATOR_KARAT"
| "SET_CALCULATOR_VALUE"
| "OPEN_SCRAP_MODAL"
| "OPEN_CUSTOM_RATES"
| "NAVIGATE_TO"
| "FILTER_DATABASE"
| "HIGHLIGHT_COIN"
| "SHOW_DIAMOND_GRADE"
| "PULSE_ELEMENT";
payload: any;
}
interface BrainContextType {
// Current awareness
awareness: BrainAwareness;
// Update brain's knowledge (called by components)
updateAwareness: (update: Partial<BrainAwareness>) => void;
// Log an action so the brain knows what just happened
logAction: (type: string, detail: string) => void;
// AI panel state
isOpen: boolean;
openBrain: (prefilledMessage?: string) => void;
closeBrain: () => void;
// Pre-filled message for when brain is opened with context
prefilledMessage: string;
setPrefilledMessage: (msg: string) => void;
// Action queue — the brain sends actions back to the app
pendingAction: BrainAction | null;
dispatchAction: (action: BrainAction) => void;
clearAction: () => void;
// Proactive suggestion the brain wants to surface
suggestion: { message: string; context: string } | null;
setSuggestion: (s: { message: string; context: string } | null) => void;
}
const BrainContext = createContext<BrainContextType | null>(null);
export function BrainProvider({ children }: { children: ReactNode }) {
const [awareness, setAwareness] = useState<BrainAwareness>({
currentPage: "/",
});
const [isOpen, setIsOpen] = useState(false);
const [prefilledMessage, setPrefilledMessage] = useState("");
const [pendingAction, setPendingAction] = useState<BrainAction | null>(null);
const [suggestion, setSuggestion] = useState<{ message: string; context: string } | null>(n
const updateAwareness = useCallback((update: Partial<BrainAwareness>) => {
setAwareness(prev => ({
...prev,
...update,
// Deep merge nested objects
calculator: update.calculator ? { ...prev.calculator, ...update.calculator } : prev.cal
diamonds: update.diamonds ? { ...prev.diamonds, ...update.diamonds } : prev.diamonds,
database: update.database ? { ...prev.database, ...update.database } : prev.database,
market: update.market ? { ...prev.market, ...update.market } : prev.market,
}));
}, []);
const logAction = useCallback((type: string, detail: string) => {
setAwareness(prev => ({
...prev,
lastAction: { type, detail, timestamp: Date.now() }
}));
}, []);
const openBrain = useCallback((prefilledMessage?: string) => {
if (prefilledMessage) setPrefilledMessage(prefilledMessage);
setIsOpen(true);
}, []);
const closeBrain = useCallback(() => {
setIsOpen(false);
setPrefilledMessage("");
}, []);
const dispatchAction = useCallback((action: BrainAction) => {
setPendingAction(action);
}, []);
const clearAction = useCallback(() => {
setPendingAction(null);
}, []);
return (
<BrainContext.Provider value={{
awareness,
updateAwareness,
logAction,
isOpen,
openBrain,
closeBrain,
prefilledMessage,
setPrefilledMessage,
pendingAction,
dispatchAction,
clearAction,
suggestion,
setSuggestion,
}}>
{children}
</BrainContext.Provider>
);
}
export function useBrain() {
const ctx = useContext(BrainContext);
if (!ctx) throw new Error("useBrain must be used inside BrainProvider");
return ctx;
}
