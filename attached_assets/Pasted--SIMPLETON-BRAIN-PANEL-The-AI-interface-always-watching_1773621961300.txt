function getContextSummary(awareness: BrainAwareness): string {
if (awareness.calculator && awareness.calculator.selectedMetal) {
const { selectedMetal, selectedKarat, display, livePrice } = awareness.calculator;
const parts = [];
if (display && display !== "0") parts.push(`${display}${awareness.calculator.selectedUnit

if (livePrice) parts.push(`$${livePrice.toFixed(0)}/oz live`);

function getPageLabel(path: string): string {
if (path === "/" || path.includes("calculator")) return "Calculator";
if (path.includes("diamond")) return "Diamonds";
if (path.includes("database")) return "Database";
if (path.includes("watches")) return "Watches";
if (path.includes("portfolio")) return "Portfolio";
if (path.includes("education")) return "Education";
return "App";
}
async function compressImage(file: File): Promise<string> {
return new Promise((resolve) => {
const canvas = document.createElement("canvas");
const ctx = canvas.getContext("2d");
const img = new Image();
img.onload = () => {
let { width, height } = img;
const max = 2048;
if (width > max || height > max) {
if (width > height) { height = (height * max) / width; width = max; }
else { width = (width * max) / height; height = max; }
}
canvas.width = width;
canvas.height = height;
ctx?.drawImage(img, 0, 0, width, height);
let q = 0.9;
let result = canvas.toDataURL("image/jpeg", q);
while (result.length > 4000 * 1024 * 1.37 && q > 0.1) {
q -= 0.1;
result = canvas.toDataURL("image/jpeg", q);
}
// Strip data URL prefix for API
resolve(result.replace(/^data:image\/\w+;base64,/, ""));
};
const reader = new FileReader();
reader.onload = () => { img.src = reader.result as string; };
reader.readAsDataURL(file);
});
